path = Path(".")
folders = [f"CAT_0{i}" for i in range(7)]

def get_position(x): 
    loc_list = np.genfromtxt(Path(f"{str(x)}.cat"))
    positions = [loc_list[[i, i+1]] for i in range(1, 7, 2)]
    return tensor(positions)

def get_items(path):
    fnames = get_image_files(path, folders=folders)
    return fnames

cats = DataBlock(
    blocks=(ImageBlock, PointBlock),
    get_items=get_items,
    get_y=get_position,
    splitter=RandomSplitter(),
    item_tfms=Resize(300, method="squish"),
    batch_tfms=Normalize.from_stats(*imagenet_stats)
)
dls = cats.dataloaders(path, bs=32)

dls.show_batch()

learn = cnn_learner(dls, resnet18, y_range=(-1, 1))
learn.lr_find()
