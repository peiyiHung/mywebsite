from fastai.vision.all import *
import warnings
warnings.filterwarnings("ignore")

Prepare for the data 

path = untar_data(URLs.CIFAR)

path.ls()

(#3) [Path('/root/.fastai/data/cifar10/labels.txt'),Path('/root/.fastai/data/cifar10/test'),Path('/root/.fastai/data/cifar10/train')]
dblock = DataBlock(blocks=(ImageBlock, CategoryBlock), 
                   get_items=get_image_files,
                   get_y=parent_label,
                   splitter=RandomSplitter(),
                   item_tfms=RandomResizedCrop(256, min_scale=0.3),
                   batch_tfms=[*aug_transforms(size=96),
                               Normalize])
dls = dblock.dataloaders(path/"train", bs=128)

dls.show_batch(max_n=12)

test_dl = dls.test_dl(get_image_files(path/"test"), with_labels=True)

learn = cnn_learner(dls, resnet50, metrics=accuracy)
